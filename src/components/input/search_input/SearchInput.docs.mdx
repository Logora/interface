import { SearchInput } from './SearchInput';
import { ConfigProvider } from '@logora/debate.context.config_provider'

An input to handle search queries with custom placeholder.

### Component usage

```js
import SearchInput from '@logora/debate.input.search_input';
```

```js
<ConfigProvider>
    <SearchInput 
        placeholder="Search"
        onSearchSubmit={(query) => console.log("SearchInput callback")} 
        disabled={false}
        reducedByDefault={false}
    />
</ConfigProvider>
```
