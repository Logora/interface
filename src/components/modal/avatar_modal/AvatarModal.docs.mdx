import { AvatarModal } from "./AvatarModal";
import { ModalProvider } from "@logora/debate.dialog.modal";
import { IntlProvider } from "react-intl";

A modal component that allows users to select an avatar picture from a library of available images.

### Component usage

```js
import { AvatarModal } from '@logora/debate.modal.avatar_modal';
```

```js
<ModalProvider>
    <IntlProvider locale="en">
        <AvatarModal
            onChooseAvatar={handleChooseAvatar}
            avatarUrlList={avatarUrlList}
            userName={"User Name"}
        />
    </IntlProvider>
</ModalProvider>
```
