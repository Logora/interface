import { ConfirmModal } from './ConfirmModal';
import { ModalProvider } from '@logora/debate.dialog.modal';

A modal component that asks for user's confirmation. The modal shows two buttons, one to confirm, one to cancel.

### Component usage
```js
    <ModalProvider>
        <ConfirmModal 
            title="User confirmation"
            question="Are you sure you want to start this debate ?"
            confirmLabel="Yes"
            cancelLabel="No"
            onConfirmCallback={() => alert("Debate started !")}
            onCancelCallback={() => alert("Debate cancelled !")}
        />
    </ModalProvider>
```
