import { BoxSkeleton } from './BoxSkeleton';

A skeleton for box loading.

### Component usage
```js
import BoxSkeleton from '@logora/debate.skeleton.box_skeleton';
```
```js
<BoxSkeleton />
```

### Properties
| **Name**      | **Type** | **Required** | **Default** | **Description** |
| ----------- | ----------- | :-----------: | ----------- | ----------- |
| <font color='#4646c6'>enableAnimation</font> | <font color="#397300">boolean</font> | &cross; | true | <font color="grey">*Enable skeleton animation*</font> |
| <font color='#4646c6'>boxHeight</font> | <font color="#397300">integer</font> | &cross; | 200 | <font color="grey">*Height of the main box in pixels*</font> |
| <font color='#4646c6'>onlyEdgeBox</font> | <font color="#397300">boolean</font> | &cross; | false | <font color="grey">*Display only the main edge box, without others elements*</font> |
| <font color='#4646c6'>noGlobalMargin</font> | <font color="#397300">boolean</font> | &cross; | false | <font color="grey">*Remove global margin*</font> |
