import { UserContentSkeleton } from './UserContentSkeleton';

A skeleton for user content loading.

### Component usage
```js
import UserContentSkeleton from '@logora/debate.skeleton.user_content_skeleton';
```
```js
<UserContentSkeleton>
    <div>Loading</div>
</UserContentSkeleton>
```

### Properties
| **Name**      | **Type** | **Required** | **Default** | **Description** |
| ----------- | ----------- | :-----------: | ----------- | ----------- |
| <font color='#4646c6'>enableAnimation</font> | <font color="#397300">boolean</font> | &cross; | true | <font color="grey">*Enable skeleton animation*</font> |
| <font color='#4646c6'>hideBody</font> | <font color="#397300">boolean</font> | &cross; | false | <font color="grey">*Hide skeleton body*</font> |
| <font color='#4646c6'>hideBody</font> | <font color="#397300">boolean</font> | &cross; | false | <font color="grey">*Hide skeleton body*</font> |
| <font color='#4646c6'>children</font> | <font color="#397300">React component</font> | &cross; | false | <font color="grey">*Children displayed at the center of the skeleton*</font> |
