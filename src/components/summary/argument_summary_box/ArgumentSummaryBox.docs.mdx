description: A component that renders an argument box with a gauge.

import { ArgumentSummaryBox } from './ArgumentSummaryBox';

### Component usage
```js
import { ArgumentSummaryBox } from '@logora/debate.summary.argument_summary_box';
```
```js
<ArgumentSummaryBox
    label={"Recurrence of the argument"}
    text={"1. The term Chocolatine is more familiar and easier to pronounce for French speakers. "}
    gauge={3}
    color={"blue"}
    tag={"Yes"}
/>
```
