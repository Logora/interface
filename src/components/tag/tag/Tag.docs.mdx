import { Tag } from './Tag';

A component that renders a tag. Text is capitalized by default.

### Component usage
```js
    import { Tag } from '@logora/debate.tag.tag';
```
```js
<Tag 
    data-tid={"tracking-id"} 
    text={"my text"}
    active={true}
    className={classNameToPass}
/>
```
