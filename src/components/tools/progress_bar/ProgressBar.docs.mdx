import { ProgressBar } from './ProgressBar';

A component that renders a progress bar. It can also render a title, and a subtitle (right side text above the progress bar).

### Component usage
```js
import ProgressBar from "@logora/debate.tools.progress_bar";
```
```js
<ProgressBar 
    progress={25} 
    goal={100} 
    title={"My title"}
    className={styles.myClass} 
    innerClassName={styles.myInnerClass}
    subtitleClassName={styles.mySubtitleClassName}
    textClassName={styles.myTextClass}
    showPercentageSubtitle={false}
    showProgressSubtitle={true}
    progressUnit={"votes"}
    barFull={false}
    isLeft={false}
/>
```
